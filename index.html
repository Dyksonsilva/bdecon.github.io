IMF API with Python 


# Example: loading IFS into pandas from API

# Import libraries
import requests
import pandas as pd

# URL for the IMF JSON Restful Web Service, 
# IFS database, and Australian export prices series
baseurl = 'http://dataservices.imf.org/REST/SDMX_JSON.svc/CompactData/'
series_request = 'IFS/Q.AU.PXP_IX.?startPeriod=1957&endPeriod=2016'
url = baseurl + series_request

# Get data from the above URL using the requests package
data = requests.get(url).json()

# Load data into a pandas dataframe and set index
au_xp = pd.DataFrame(data['CompactData']['DataSet']['Series']['Obs'])
au_xp = au_xp.set_index(pd.to_datetime(au_xp['@TIME_PERIOD']))

# Basic plot 
au_xp['@OBS_VALUE'].astype(float).plot()
